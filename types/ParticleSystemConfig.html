    OptionalstartLifetime

    startLifetime?: Constant | RandomBetweenTwoConstants | LifetimeCurve

    Default

    5.0

        { x: 0, y: 0.275, percentage: 0 },

      scale: 2
    };

    // Easing curve example with scaling.
    startLifetime: {
      type: LifeTimeCurve.EASING,

      scale: 0.5
